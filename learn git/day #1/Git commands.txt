# Инициализация гита в текущую репозиторию
git init

# Статус гита
git status

# Добавления конкретного файла или папки в индекс гита
git add <Путь конкретного файла или папки>

# Добавления всех файлов и папок которые есть в списке не добавленных в индекс
git add .

# Добавление коммитов
git commit -m "<Описание коммитов>"

# Отправка коммитов на удаленный репозиторий
git push -u origin <название текущей ветки>

# Отправка последующих коммитов
git push

# Список веток
git branch

# Смена ветки
git checkout <название ветки>

# Создание ветки
git checkout -b <название ветки>

# Подтянуть изменения из удаленного репозитория
git pull

# Мердж веток
git merge <название ветки>
